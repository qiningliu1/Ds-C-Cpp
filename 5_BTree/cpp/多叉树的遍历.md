#### **Vector容器核心篇**

**🤔 vector是什么？与C数组有何区别？**

vector是C++标准库提供的动态数组容器，自动管理内存。

```cpp
#include <vector>
using namespace std;

// 基本声明和初始化
vector<int> nums;                    // 空vector
vector<int> nums2(5);                // 包含5个0的vector
vector<int> nums3(5, 10);            // 包含5个10的vector
vector<int> nums4{1, 2, 3, 4, 5};    // 初始化列表
vector<Node*> children;              // 指针vector
```

**🤔 vector的核心方法有哪些？**

```cpp
vector<int> vec;

// 添加和访问
vec.push_back(10);           // 尾部添加
vec.push_back(20);
vec.push_back(30);

cout << vec[0] << endl;      // 索引访问：10
cout << vec.back() << endl;  // 最后元素：30
cout << vec.front() << endl; // 第一个元素：10

// 大小和容量
cout << vec.size() << endl;     // 当前元素数量：3
cout << vec.capacity() << endl; // 内存容量：>=3
cout << vec.empty() << endl;    // 是否为空：false

// 插入和删除
vec.insert(vec.begin() + 1, 15);  // 位置1插入15
vec.erase(vec.begin());           // 删除第一个元素
vec.clear();                      // 清空所有元素
```

**🤔 vector的三种遍历方式？**

```cpp
vector<int> vec{1, 2, 3, 4, 5};

// 方式1：传统索引遍历
for (int i = 0; i < vec.size(); i++) {
    cout << vec[i] << " ";
}

// 方式2：迭代器遍历
for (auto it = vec.begin(); it != vec.end(); ++it) {
    cout << *it << " ";
}

// 方式3：范围for循环（C++11推荐）
for (int num : vec) {
    cout << num << " ";
}
```

#### **类方法设计篇**

**🤔 类内方法和类外方法有什么区别？**

```cpp
class Node {
private:
    int val;
    vector<Node*> children;
    
public:
    // 类内方法：默认inline候选
    void addChild(Node* child) {
        if (child != nullptr) {
            children.push_back(child);
        }
    }
    
    // 只声明，在类外定义
    void complexTraverse();
    void deepCopy(const Node& other);
};

// 类外定义：需要使用作用域解析符::
void Node::complexTraverse() {
    // 复杂的遍历逻辑，不适合inline
    if (this == nullptr) return;
    
    cout << "访问节点: " << val << endl;
    
    for (Node* child : children) {
        child->complexTraverse();
    }
    
    cout << "离开节点: " << val << endl;
}

void Node::deepCopy(const Node& other) {
    // 复杂的深拷贝逻辑
    this->val = other.val;
    this->children.clear();
    
    for (Node* child : other.children) {
        Node* newChild = new Node(child->val);
        newChild->deepCopy(*child);
        this->children.push_back(newChild);
    }
}
```

**🤔 什么时候应该使用类内方法，什么时候使用类外方法？**

- **类内方法适用场景**：
  - 简单的getter/setter方法
  - 一行或几行的简单逻辑
  - 频繁调用的小函数（inline优化）

- **类外方法适用场景**：
  - 复杂的算法逻辑
  - 代码超过10行的方法
  - 需要包含大量头文件的方法
  - 编译时间优化考虑

#### **范围for循环与成员访问篇**

**🤔 类成员函数中如何访问成员变量？**

```cpp
class Node {
private:
    int val;
    vector<Node*> children;
    
public:
    void traverse() {
        // 方式1：直接访问成员变量
        for (Node* child : children) {
            child->traverse();
        }
        
        // 方式2：显式使用this指针
        for (Node* child : this->children) {
            child->traverse();
        }
        
        // 两种方式等价，第一种更简洁
    }
    
    void printInfo() {
        cout << "节点值: " << val << endl;          // 直接访问
        cout << "子节点数: " << children.size() << endl;  // 直接访问
        cout << "节点值: " << this->val << endl;    // 显式this
    }
};
```

**🤔 范围for循环的底层实现原理？**

```cpp
// 用户写的代码
for (Node* child : children) {
    child->traverse();
}

// 编译器实际生成的代码（简化版）
for (auto it = children.begin(); it != children.end(); ++it) {
    Node* child = *it;
    child->traverse();
}
```

**🤔 范围for循环的不同变体？**

```cpp
vector<Node*> children;

// 1. 值拷贝（如果元素大，开销大）
for (Node* child : children) {
    // child是children中元素的拷贝
}

// 2. 引用（推荐，避免拷贝）
for (Node*& child : children) {
    // child是children中元素的引用，可以修改
    child = new Node(10);  // 可以修改原容器中的指针
}

// 3. const引用（只读，避免拷贝）
for (const Node* child : children) {
    // child是只读的，不能修改指针指向
    child->traverse();  // 可以调用const方法
}

// 4. auto自动推导（推荐）
for (auto child : children) {        // auto推导为Node*
    child->traverse();
}

for (auto& child : children) {       // auto推导为Node*&
    child = new Node(20);
}

for (const auto& child : children) { // auto推导为const Node*
    child->traverse();
}
```

#### **C++与C语言对比篇**

**🤔 内存管理方式的根本区别？**

```cpp
// C语言方式：手动管理一切
typedef struct Node {
    int val;
    struct Node** children;
    int childCount;
    int capacity;
} Node;

Node* createNode(int val) {
    Node* node = malloc(sizeof(Node));
    if (node == NULL) return NULL;
    
    node->val = val;
    node->children = malloc(sizeof(Node*) * 4);
    if (node->children == NULL) {
        free(node);
        return NULL;
    }
    node->childCount = 0;
    node->capacity = 4;
    return node;
}

void addChild(Node* parent, Node* child) {
    if (parent->childCount >= parent->capacity) {
        // 手动扩容
        parent->capacity *= 2;
        Node** newChildren = realloc(parent->children, 
                                   sizeof(Node*) * parent->capacity);
        if (newChildren == NULL) {
            // 扩容失败处理
            return;
        }
        parent->children = newChildren;
    }
    parent->children[parent->childCount++] = child;
}

void freeNode(Node* node) {
    if (node == NULL) return;
    
    // 递归释放子节点
    for (int i = 0; i < node->childCount; i++) {
        freeNode(node->children[i]);
    }
    
    free(node->children);
    free(node);
}

// C++方式：自动管理，异常安全
class Node {
private:
    int val;
    vector<Node*> children;  // 自动管理数组内存
    
public:
    Node(int v) : val(v) {
        // 构造函数，vector自动初始化
    }
    
    ~Node() {
        // 析构函数：自动清理
        for (Node* child : children) {
            delete child;
        }
        // vector自动释放内存
    }
    
    void addChild(Node* child) {
        children.push_back(child);  // 自动扩容，异常安全
    }
    
    // 拷贝构造函数（深拷贝）
    Node(const Node& other) : val(other.val) {
        for (Node* child : other.children) {
            children.push_back(new Node(*child));  // 递归深拷贝
        }
    }
    
    // 赋值操作符
    Node& operator=(const Node& other) {
        if (this != &other) {
            // 清理原有数据
            for (Node* child : children) {
                delete child;
            }
            children.clear();
            
            // 深拷贝新数据
            val = other.val;
            for (Node* child : other.children) {
                children.push_back(new Node(*child));
            }
        }
        return *this;
    }
};
```

#### **现代C++最佳实践篇**

**🤔 如何设计现代C++的多叉树？**

```cpp
#include <memory>
#include <vector>
#include <queue>
#include <iostream>

// 使用智能指针的现代C++实现
class ModernMultiTree {
private:
    struct Node {
        int val;
        vector<unique_ptr<Node>> children;
        
        Node(int v) : val(v) {}
        
        // 移动构造函数
        Node(Node&& other) noexcept 
            : val(other.val), children(std::move(other.children)) {}
        
        // 移动赋值操作符
        Node& operator=(Node&& other) noexcept {
            if (this != &other) {
                val = other.val;
                children = std::move(other.children);
            }
            return *this;
        }
        
        // 禁用拷贝构造和拷贝赋值
        Node(const Node&) = delete;
        Node& operator=(const Node&) = delete;
    };
    
    unique_ptr<Node> root;

public:
    // 构造函数
    ModernMultiTree(int rootVal) 
        : root(make_unique<Node>(rootVal)) {}
    
    // 添加子节点
    void addChild(int parentVal, int childVal) {
        Node* parent = findNode(root.get(), parentVal);
        if (parent) {
            parent->children.push_back(make_unique<Node>(childVal));
        }
    }
    
    // 批量添加子节点
    void addChildren(int parentVal, const vector<int>& childValues) {
        Node* parent = findNode(root.get(), parentVal);
        if (parent) {
            for (int val : childValues) {
                parent->children.push_back(make_unique<Node>(val));
            }
        }
    }
    
    // 深度优先遍历
    void dfsTraverse() const {
        if (root) {
            dfsHelper(root.get());
        }
    }
    
    // 广度优先遍历
    void bfsTraverse() const {
        if (!root) return;
        
        queue<const Node*> q;
        q.push(root.get());
        
        while (!q.empty()) {
            const Node* current = q.front();
            q.pop();
            
            cout << current->val << " ";
            
            for (const auto& child : current->children) {
                q.push(child.get());
            }
        }
        cout << endl;
    }
    
    // 获取树的深度
    int getDepth() const {
        return root ? getDepthHelper(root.get()) : 0;
    }
    
    // 获取节点数量
    int getNodeCount() const {
        return root ? getNodeCountHelper(root.get()) : 0;
    }
    
private:
    Node* findNode(Node* node, int val) {
        if (!node) return nullptr;
        if (node->val == val) return node;
        
        for (auto& child : node->children) {
            Node* result = findNode(child.get(), val);
            if (result) return result;
        }
        return nullptr;
    }
    
    void dfsHelper(const Node* node) const {
        if (!node) return;
        
        cout << node->val << " ";
        
        for (const auto& child : node->children) {
            dfsHelper(child.get());
        }
    }
    
    int getDepthHelper(const Node* node) const {
        if (!node) return 0;
        
        int maxDepth = 0;
        for (const auto& child : node->children) {
            maxDepth = max(maxDepth, getDepthHelper(child.get()));
        }
        
        return maxDepth + 1;
    }
    
    int getNodeCountHelper(const Node* node) const {
        if (!node) return 0;
        
        int count = 1;  // 当前节点
        for (const auto& child : node->children) {
            count += getNodeCountHelper(child.get());
        }
        
        return count;
    }
};

// 使用示例
void demonstrateModernTree() {
    ModernMultiTree tree(1);
    
    // 添加子节点
    tree.addChild(1, 2);
    tree.addChild(1, 3);
    tree.addChild(1, 4);
    
    // 批量添加
    tree.addChildren(2, {5, 6});
    tree.addChildren(3, {7});
    
    cout << "DFS遍历: ";
    tree.dfsTraverse();
    
    cout << "BFS遍历: ";
    tree.bfsTraverse();
    
    cout << "树的深度: " << tree.getDepth() << endl;
    cout << "节点数量: " << tree.getNodeCount() << endl;
}
```

## 📋 总结区域 (Summary)

**核心要点：**
1. **vector容器**：C++的动态数组，自动内存管理，丰富的操作方法
2. **类方法设计**：简单方法类内定义（inline），复杂方法类外定义
3. **范围for循环**：现代C++推荐的遍历方式，简洁高效
4. **引用传递**：避免不必要的拷贝，const引用保证只读
5. **const正确性**：函数不修改对象状态时应声明为const
6. **现代C++**：智能指针、移动语义、auto等新特性提升代码质量

**C++相比C的优势：**
- **内存管理**：RAII机制，构造函数/析构函数自动管理资源
- **参数传递**：引用传递避免拷贝开销
- **容器支持**：STL容器自动扩容、异常安全
- **类型安全**：编译时检查，减少运行时错误
- **代码简洁**：范围for、auto、lambda等特性提升可读性

**最佳实践建议：**
- 优先使用vector而非原始数组
- 大对象传递使用const引用
- 不修改对象状态的函数声明为const
- 使用智能指针管理动态内存
- 利用现代C++特性提升代码质量

**记忆口诀：**
- "vector动态自管理，push_back添加很方便"
- "引用传递零拷贝，const修饰保安全"
- "范围for循环真简洁，auto推导类型全"
- "智能指针管内存，RAII机制保平安"
